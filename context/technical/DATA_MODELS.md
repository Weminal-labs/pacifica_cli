# Data Models

> Generated by setup wizard on 2026-03-29. Will be refined during build.

## Config (`.pacifica.yaml`)

```typescript
interface PacificaConfig {
  network: 'testnet' | 'mainnet';
  private_key: string;     // Base58-encoded Ed25519 secret key (Solana wallet)
  defaults: {
    leverage: number;       // default: 5
    tp_distance: number;    // default: 3 (percent)
    sl_distance: number;    // default: 2 (percent)
    slippage: number;       // default: 0.5 (percent)
  };
  agent: AgentConfig;
  hooks: HooksConfig;
}

interface AgentConfig {
  enabled: boolean;
  daily_spending_limit: number;
  max_order_size: number;
  max_leverage: number;
  allowed_actions: string[];
  blocked_actions: string[];
  require_confirmation_above: number;
}

interface HooksConfig {
  on_fill?: string[];
  on_liquidation_warning?: string[];
  on_funding?: string[];
  on_smart_order_trigger?: string[];
  on_large_pnl_change?: string[];
  on_position_open?: string[];
  on_position_close?: string[];
}
```

## Market Data (from Pacifica API)

> All numeric values come as strings from the API and are parsed to numbers in the SDK.

```typescript
interface Market {
  symbol: string;          // "BTC", "ETH", "SOL" (not "ETH-PERP")
  price: number;
  change_24h: number;      // percent
  volume_24h: number;      // USD
  open_interest: number;   // USD
  funding_rate: number;    // current period rate
}

interface OrderBook {
  symbol: string;
  bids: [number, number][];  // [price, size]
  asks: [number, number][];
}
```

## Trading (from Pacifica API)

```typescript
interface Order {
  orderId: number;
  symbol: string;
  side: 'bid' | 'ask';
  type: 'market' | 'limit';
  price?: number;            // for limit orders
  initialAmount: number;     // original order size
  filledAmount: number;      // how much has been filled
  cancelledAmount: number;   // how much has been cancelled
  status: 'open' | 'filled' | 'partially_filled' | 'cancelled';
  created_at: string;
}

interface Position {
  symbol: string;
  side: 'long' | 'short';
  size: number;
  entry_price: number;
  liquidation_price: number;
  leverage: number;
  margin: number;
  // mark_price and unrealized_pnl are calculated client-side, not returned by API
}

interface Account {
  balance: number;
  available_margin: number;
  used_margin: number;
}
```

## Local Storage Models

### Journal Entry (`.pacifica/journal.json`)

```typescript
interface JournalEntry {
  id: string;              // UUID
  timestamp: string;       // ISO
  type: 'fill' | 'position_close' | 'smart_order_trigger';
  symbol: string;
  side: 'buy' | 'sell' | 'long' | 'short';
  size: number;
  price: number;

  fees: number;
  leverage: number;

```

### Smart Order State (`.pacifica/smart-orders.json`)

```typescript
interface SmartOrder {
  id: string;
  type: 'trailing_stop' | 'partial_tp' | 'breakeven' | 'dca' | 'time_stop';
  symbol: string;
  created_at: string;
  status: 'active' | 'triggered' | 'cancelled';
  config: TrailingStopConfig | PartialTpConfig | DcaConfig | TimeStopConfig;
  last_checked?: string;
}

interface TrailingStopConfig {
  distance: number;        // percent or dollar
  distance_type: 'percent' | 'dollar';
  highest_price?: number;  // tracked high (for longs)
  lowest_price?: number;   // tracked low (for shorts)
}

interface PartialTpConfig {
  levels: { price: number; percent: number }[];
  filled_levels: number[];
}
```

## Conventions

- All timestamps are ISO 8601 strings
- All prices and amounts from API arrive as strings, parsed to numbers in the SDK
- Currency amounts are in USD
- Sizes are in the base asset unit (e.g., ETH, SOL, BTC)
- Funding rates are per-period decimals (0.01 = 1%)
- JSON files are append-only (except smart-orders which is read-write)
- UUIDs use `crypto.randomUUID()`
